#include <iostream>
#include <string>

using namespace std; 

class Besch
{
  private: string name;
  public:  Besch(string name):name(name){}
  string getName(){return name;}
  void setName(string name){this->name=name;}
  virtual void display(ostream& os){os<<name;}
  virtual double calc()=0;

};

class LohnArb : public Besch
{
  private:
  int Stunden;
  double stdLohn;
  public:
  LohnArb(string name, double stdLohn):Besch(name)
  {this->stdLohn=stdLohn;}
  double getStdLohn(){return stdLohn;}
  int getStdunden(){return Stunden;}
  void setStdLohn(double stdLohn){this->stdLohn=stdLohn;}
  void setStunden(int Stunden){this->Stunden=Stunden;}
  //void display(ostream &os){os<<getName()<<": "<<stdLohn;}
  virtual void display(ostream &os){Besch::display(os); os<<": "<<stdLohn;}
  virtual double calc(){return stdLohn*Stunden;}
};

class Angest : public Besch
{
  private:
  double Gehalt;
  public:
  Angest(string name, double Gehalt):Besch(name)
  {this->Gehalt=Gehalt;}
  double getGehalt(){return Gehalt;}
  void setGehalt(double Gehalt){this->Gehalt=Gehalt;}
  virtual void display(ostream &os){os<<getName()<<": "<<Gehalt;}
  //void display(ostream &os){Besch::display(os); os<<": "<<stdLohn;}
  virtual double calc(){return Gehalt;}
};

class Haendler:public LohnArb
{
  double prov;
  double Umsatz;
public:
  Haendler(string name, double stdLohn, double prov):LohnArb(name,stdLohn)
  {this->prov=prov;}
  double getProv(){return prov;}
  double getUmsatz(){return Umsatz;}
  void setProv(double prov){this->prov=prov;}
  void setUmsatz(double Umsatz){this->Umsatz=Umsatz;}
  //void display(ostream &os){os<<getName()<<": "<<stdLohn;}
  virtual void display(ostream &os){LohnArb::display(os); os<<": "<<prov;}
  virtual double calc(){return LohnArb::calc()+Umsatz*prov;}
};

int main()
{
  LohnArb b1("Lehmann",8.02);
  b1.setStunden(22);
  b1.display(cout);
  cout<<endl; 
  LohnArb b2("Huckebein",8.52);
  b2.setStunden(22);
  b2.display(cout);
  cout<<" Lohn:"<<b2.calc()<<endl; 
  Angest  b3("Meier",3243.03);
  b3.display(cout);
  cout<<" Gehalt:"<<b3.calc()<<endl; 
  Haendler b4("Ohlsen",9.88,0.12);
  b4.setStunden(21);
  b4.setUmsatz(21975.95);
  b4.display(cout);
  cout<<" Lohn+Prov:"<<b4.calc()<<endl;
  cout<<endl;
  
  Besch* Belegschaft[]={&b1,&b2,&b3,&b4};
  for (int i=0; i<4;i++)
  {
    Belegschaft[i]->display(cout);
    cout<<": "<<Belegschaft[i]->calc()<<endl;
    cout<<endl;
  }
  
  
}
