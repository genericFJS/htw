template <class T>
IntArr<T> :: IntArr (int Sz){
	Size = Sz;
	IA = new T [Size];
	for (int i=0; i<Sz; ++i) IA[i] = 0;
}

template <class T>
IntArr<T> :: IntArr (const IntArr<T>& Other){
	Size = Other.Size;
	IA = new T[Size];
	for (int i=0; i<Size; i++) IA[i]=Other.IA[i];
