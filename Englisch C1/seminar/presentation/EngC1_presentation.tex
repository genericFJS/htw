\section*{Generic programming}

\begin{frame}
\begin{itemize}
\setlength\itemsep{3em}
\item easy to use % you write what you want to have, without a concept
\item limited scalability % you have limited concepts to describe complex programms
\end{itemize}
\end{frame}

\begin{frame}
(Exaggerated) example:\bigskip\\
Programming a Human being:\medskip\\
Eating as …
\begin{itemize}
\setlength\itemsep{1em}
\item … an adult
\item … a student
\item … a child
\item … a baby
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\begin{lstlisting}[language=C]
if ( adult ) {
	cook_something();
	eat();
} else if ( student ) {
	call_pizza_service();
	wait();
	eat();
} else if ( child ) {
	tell_mom_youre_hungry();
	wait();
	eat_only_dessert();
} else if ( baby ) {
	while ( nothing_happens ) {
		cry();
	}
	suck_delicious_milk_from_moms_breasts();
}
\end{lstlisting}
\end{frame}

\end{frame}

\section*{Inheritance}

\begin{frame}
\begin{center}
\resizebox{.5\textwidth}{!}{
\begin{tikzpicture}

\umlVHVinherit{Student}{Human} 
\end{tikzpicture}
}
\end{center}
\begin{itemize}
\setlength\itemsep{1em}
\item Adult uses unchanged function \\(or may implement a new based on \emph{time\_available})
\item Student implements new function \\(and may use \emph{parent}-function)
\end{itemize}
\end{frame}

\section*{Polymorphism}
\begin{frame}
\begin{itemize}
\setlength\itemsep{1em}
\item easy to expand upon
\item \emph{child}-classes use
\begin{itemize}
\item parent properties
\item parent functions
\end{itemize}
$\Rightarrow$ \textbf{Inheritance}
\item \emph{child}-classes can change behavior of
\begin{itemize}
\item parent properties
\item parent functions
\end{itemize}
$\Rightarrow$ \textbf{Polymorphism}
\end{itemize}
\end{frame}
\begin{frame}
